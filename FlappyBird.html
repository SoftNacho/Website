Presentation of Project

My partner and I chose to make our version of the famous video game Flappy Bird using a VGA monitor to display the graphics, as well as a keyboard to use as the input device. The main idea for this game is that, using the flappy_bird.sv module, we are forcing a bird to bounce up and down using the keyboard in order to avoid some obstacles (represented as pipes) that move from right to left (as if the bird was actually moving from left to right). We used sprites to display the pipes as well as the bird and the background after having drawn them on Microsoft Paint. A Python script was utilized to convert the MS Paint PNG’s to text files which were integrated with Quartus via ROM’s/sprites. The programming languages used for our were project were SystemVerilog (majority of the modules), C, and Python.

module flappy_bird(input 			Reset, Clk, Space, reset, go, pause, dead,
						 input  [15:0] keycode,
						 input  [9:0]  PipeL_X, PipeR_X,
											Pipe_Space_LT, Pipe_Space_LB,  
											Pipe_Space_RT, Pipe_Space_RB, 
											PipeW,
						 output [9:0]  BirdX, BirdY, BirdW_out, BirdH_out,
						 output 			hit
	 );

    logic [9:0] Bird_X_Pos, Bird_X_Motion, Bird_Y_Pos, Bird_Y_Motion, Bird_Size;
	 logic [9:0] BirdW, BirdH;
    logic bird_clk;

	assign Bird_Size = 10'd50;
	assign BirdW = 10'd50;
	assign BirdH = 10'd50;

    //using 600x480 screen 
    	//setting bird position on center of screen for initialization and reset
    parameter [9:0] Bird_X_Init_Reset = 10'd295;  // Middle on X
    parameter [9:0] Bird_Y_Init_Reset = 10'd215;  // Middle on Y

    parameter [9:0] Bird_X_Left_Most = 10'd0;     // Leftmost on X
    parameter [9:0] Bird_X_Right_Most = 10'd639;  // Rightmost on X

    parameter [9:0] Bird_Y_Top = 10'd0;       	// Topmost on Y
    parameter [9:0] Bird_Y_Bottom = 10'd479;  	// Bottommost on Y
    parameter [9:0] Bird_Y_Flap = 10'd1;    	// Step size on Y


    //getting bird to move
    always_ff @ (posedge Clk or posedge Reset)
    begin: Move_Bird
    	if (Reset)
    	begin
    		Bird_Y_Motion <= 10'd0;
    		Bird_X_Motion <= 10'd0;
    		Bird_Y_Pos <= Bird_Y_Init_Reset;
    		Bird_X_Pos <= Bird_X_Init_Reset;
  			hit = 1'b0;
    	end
		
		else if(reset)
		begin
    		Bird_Y_Motion <= 10'd0;
    		Bird_X_Motion <= 10'd0;
    		Bird_Y_Pos <= Bird_Y_Init_Reset;
    		Bird_X_Pos <= Bird_X_Init_Reset;
  			hit = 1'b0;
    	end

		else if(pause)
		begin
        Bird_Y_Motion <= 10'd0;
		  Bird_X_Motion <= 10'd0;
		end
		
		else if(dead)
		begin
        Bird_Y_Motion <= 10'd0;
		  Bird_X_Motion <= 10'd0;
		end
		
    	else 
    	begin
    		if (go == 1'b1)
    		begin
					//if (keycode[7:0] == 8'd44) // space bar?
					if(keycode[7:0] == 8'h2C || Space)
					//if(Space)
					begin
						Bird_Y_Motion <= (~(10'd4) + 1'b1);
					end

					else if(Space)
					begin
						Bird_Y_Motion <= (~(10'd4) + 1'b1);
					end
					
					else
					begin
						Bird_Y_Motion <= Bird_Y_Motion + Bird_Y_Flap;
					end
				//end
			end

			else
			begin
				Bird_Y_Motion <= 10'd0;
			end 

			Bird_Y_Pos <= (Bird_Y_Pos + Bird_Y_Motion);

		end
	end

	//updating coordinates
    assign BirdX = Bird_X_Pos;
    assign BirdY = Bird_Y_Pos;
	 assign BirdH_out = BirdH;
	 assign BirdW_out = BirdW;
	 
endmodule

module FSM(input  logic        Reset, Clk, Space, Pause, hit,
           input  logic [15:0] keycode,
           output logic        reset, go, pause, dead
  );

  // title later
  enum logic [1:0] {RESET, GO, PAUSED, DEAD} State, Next_state;

  always_ff @ (posedge Clk)
  begin
    if(Reset) State <= RESET;
    else State <= Next_state;
  end

  always_comb
  begin

	 Next_state = State;
	 
    unique case(State)
    
	 RESET:
	 begin
		
		if(keycode[7:0] == 8'h2C)     Next_state <= GO;
		else if(Space)     Next_state <= GO;
		else     Next_state <= RESET;
		
    end
	 
	 GO:    
	 begin	 
		if(Reset) 				  Next_state <= RESET;
		else if(Pause)		 	  Next_state <= PAUSED;
		else if(hit == 1'b1)   Next_state <= DEAD;
		else 					 	  Next_state <= GO;
    end

	 PAUSED:  
	 begin
		if(Reset) 							   Next_state <= RESET;
		else if(Pause)					 	   Next_state <= GO;
		else										Next_state <= PAUSED;

	 end
	 
	 DEAD:  
	 begin
		if(Reset) 							  Next_state <= RESET;
		else									  Next_state <= DEAD;
	 end
	 
	 default: ;
	 endcase

  end

  always_comb
  begin

    reset = 1'b0;
    go = 1'b0;
    pause = 1'b0;
	 dead = 1'b0;

    unique case(State)
    RESET:   reset = 1'b1;
    GO:      go    = 1'b1;
    PAUSED:	 pause = 1'b1;
	 DEAD:    dead  = 1'b1;
	 default: ;
	 endcase

  end

endmodule

module  VGA_controller (input              Clk,         // 50 MHz clock
                                           Reset,       // reset signal
                        output logic       VGA_HS,      // Horizontal sync pulse.  Active low
                                           VGA_VS,      // Vertical sync pulse.  Active low
                                           VGA_CLK,     // 25 MHz VGA clock output
                                           VGA_BLANK_N, // Blanking interval indicator.  Active low.
                                           VGA_SYNC_N,  // Composite Sync signal.  Active low.  We don't use it in this lab,
                                                        // but the video DAC on the DE2 board requires an input for it.
                        output logic [9:0] DrawX,       // horizontal coordinate
                                           DrawY        // vertical coordinate
                        );     
    
    // 800 pixels per line (including front/back porch
    // 525 lines per frame (including front/back porch)
    parameter [9:0] H_TOTAL = 10'd800;
    parameter [9:0] V_TOTAL = 10'd525;
    
    logic VGA_HS_in, VGA_VS_in, VGA_BLANK_N_in;
    logic [9:0] h_counter, v_counter;
    logic [9:0] h_counter_in, v_counter_in;
    
    assign VGA_SYNC_N = 1'b0;
    assign DrawX = h_counter;
    assign DrawY = v_counter;

    // Generate VGA_CLK
    always_ff @ (posedge Clk)
    begin
        if (Reset)
        begin
            VGA_CLK <= 1'b0;
        end
        else
        begin
            VGA_CLK <= ~VGA_CLK;
        end
    end
    
    // VGA control signals
    always_ff @ (posedge VGA_CLK or posedge Reset)
    begin
        if (Reset)
        begin
            VGA_HS <= 1'b0;
            VGA_VS <= 1'b0;
            VGA_BLANK_N <= 1'b0;
            h_counter <= 10'd0;
            v_counter <= 10'd0;
        end
        else
        begin
            VGA_HS <= VGA_HS_in;
            VGA_VS <= VGA_VS_in;
            VGA_BLANK_N <= VGA_BLANK_N_in;
            h_counter <= h_counter_in;
            v_counter <= v_counter_in;
        end
    end
    
    always_comb
    begin
        // horizontal and vertical counter
        h_counter_in = h_counter + 10'd1;
        v_counter_in = v_counter;
        if(h_counter + 10'd1 == H_TOTAL)
        begin
            h_counter_in = 10'd0;
            if(v_counter + 10'd1 == V_TOTAL)
                v_counter_in = 10'd0;
            else
                v_counter_in = v_counter + 10'd1;
        end
        // Horizontal sync pulse is 96 pixels long at pixels 656-752
        // (Signal is registered to ensure clean output waveform)
        VGA_HS_in = 1'b1;
        if(h_counter_in >= 10'd656 && h_counter_in < 10'd752)
            VGA_HS_in = 1'b0;
        // Vertical sync pulse is 2 lines (800 pixels each) long at line 490-491
        //(Signal is registered to ensure clean output waveform)
        VGA_VS_in = 1'b1;
        if(v_counter_in >= 10'd490 && v_counter_in < 10'd492)
            VGA_VS_in = 1'b0;
        // Display pixels (inhibit blanking) between horizontal 0-639 and vertical 0-479 (640x480)
        VGA_BLANK_N_in = 1'b0;
        if(h_counter_in < 10'd640 && v_counter_in < 10'd480)
            VGA_BLANK_N_in = 1'b1;
    end
    
endmodule

module PipeLB(input logic         Reset, Clk, reset, go, pause,
              output logic [11:0] PipeX, Loop,
				  //input int PipeX, Loop,
				  output logic [9:0]  PipeY
  );
  
  logic [9:0]  step, height;
  logic [11:0] start, x_max, x_position, Pipe_RS_X;
  

  assign Pipe_RS_X = PipeX + 12'd100;
  assign x_max = 12'd640;
  assign start = x_max;
  assign step = 10'd1;
  
  logic [9:0] Temp_Height;
  assign Temp_Height = 10'd290;


  always_ff @ (posedge Clk or posedge Reset)
  begin

    if(Reset)
    begin
		height <= Temp_Height;
      x_position <= start;
    end
	 
	 else if(reset)
	 begin
		height <= Temp_Height;
      x_position <= start;
    end

	 else if(pause)
	 begin
        x_position <= x_position;
    end

	 
	 else
	 begin
		// reset or pause state
      if(go == 1'b0)
        x_position <= x_position;
		// left edge
		else if(PipeX <= 12'd0 || PipeX >= 12'd4000)
		begin
			height <= Temp_Height;
			x_position <= 12'd1500;
		end
		// go
		else 
		begin
			height <= Temp_Height;
			x_position <= x_position - step;
		end
	 end

  end

  assign PipeX = x_position;
  assign PipeY = height;

endmodule

module Collision(input logic 			Clk, Reset, reset, go, pause, dead,
					  input logic [11:0] PipeL_X, PipeR_X, Pipe3_X, 
												Pipe4_X, Pipe5_X, Pipe6_X,
												Pipe_Space_LT, Pipe_Space_LB,
												Pipe_Space_RT, Pipe_Space_RB, 
												Pipe_Space_3T, Pipe_Space_3B,
												Pipe_Space_4T, Pipe_Space_4B,
												Pipe_Space_5T, Pipe_Space_5B,
												Pipe_Space_6T, Pipe_Space_6B,
					  input logic [9:0]  Bird_X, Bird_Y,
					  output logic       hit,
					  output logic [8:0] counter_out, pipe_count_out
	);
	
	logic [8:0] counter, pipe_count;
	
	always_ff @(posedge Clk or posedge Reset)
	begin
		
		if(Reset)
		begin
			counter <= 9'd0;
			pipe_count <= 9'd1;
			hit <= 1'b0;
		end
	 
		else if(reset || pause || dead)
		begin
			counter <= counter;
			pipe_count <= pipe_count;
			hit <= hit;
		end
	
		
		else if(go)
		begin
		
			//bird past top
			if(Bird_Y <= 10'd0) 
				hit <= 1'b1;

			//bird past bottom
			else if((Bird_Y + 10'd50) >= 10'd479)
				hit <= 1'b1;
		
			else if( (counter >= 9'd295 && pipe_count == 9'd1) || (counter >= 9'd100 && pipe_count > 9'd1 ) )
			begin
				// 1
				if( pipe_count == 9'd1 && (Bird_Y <= 12'd140 || Bird_Y >= 12'd240) )
				begin
					hit <= 1'b1;
				end
				
				// 2
				if( pipe_count == 9'd2 && (Bird_Y <= 12'd240 || Bird_Y >= 12'd340) )
				begin
					hit <= 1'b1;
				end
				
				// 3
				if( pipe_count == 9'd3 && (Bird_Y <= 12'd300 || Bird_Y >= 12'd400) )
				begin
					hit <= 1'b1;
				end
				
				// 4
				if( pipe_count == 9'd4 && (Bird_Y <= 12'd100 || Bird_Y >= 12'd200) )
				begin
					hit <= 1'b1;
				end
				
				// 5
				if( pipe_count == 9'd5 && (Bird_Y <= 12'd170 || Bird_Y >= 12'd270) )
				begin
					hit <= 1'b1;
				end
				
				// 6
				if( pipe_count == 9'd6 && (Bird_Y <= 12'd90 || Bird_Y >= 12'd190) )
				begin
					hit <= 1'b1;
				end
				
			
				if(pipe_count == 9'd1 && counter >= 9'd445)
				begin
					counter <= 9'd0;
					pipe_count <= pipe_count + 9'd1;
				end
				else if(pipe_count > 9'd1 && counter >= 9'd225) 
				begin
					counter <= 9'd0;
					pipe_count <= pipe_count + 9'd1;
				end
                
				counter <= counter + 1'd1;
			end
			
			else
			begin
				counter <= counter + 9'd1;
			end
		end

		else
		begin
			counter <= counter + 9'd1;
			hit <= hit;
		end


  end

  assign counter_out = counter;
  assign pipe_count_out = pipe_count;
	
endmodule

module Score(input logic        Clk, Reset, go, pause, dead,
             input logic [11:0] PipeL_X, PipeR_X, Pipe3_X,
										  Pipe4_X, Pipe5_X, Pipe6_X,
										  Bird_X,
             output      [3:0]  tens_out, ones_out
  );

  logic [3:0] tens, ones;
  logic [8:0] counter, pipe_count;

  
  always_ff @ (posedge Clk or posedge Reset)
  begin

    if(Reset)
    begin
		counter <= 9'd0;
		pipe_count <= 9'd0;
      tens <= 4'd0;
      ones <= 4'd0;
    end
	 
	 else if(pause || dead)
	 begin
		counter <= counter;
		pipe_count <= pipe_count;
		tens <= tens;
		ones <= ones;
	 end

    else if(go)
    begin
      if( (counter >= 9'd345 && pipe_count == 9'd0) || (counter >= 9'd250 && pipe_count > 9'd0 ) )
      begin
        if(ones != 4'd9)
        begin
          ones <= ones + 1'd1;
          tens <= tens;
        end
		  else
		  begin
          ones <= 4'd0;
          tens <= tens + 1'd1;
        end
		  counter <= 9'd0;
		  pipe_count <= pipe_count + 9'd1;
		end
      else
      begin
          counter <= counter + 9'd1;
      end
    end

    else
    begin
      ones <= ones;
      tens <= tens;
    end

  end

  assign ones_out = ones;
  assign tens_out = tens;

endmodule
