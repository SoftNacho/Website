Presentation of Project

For this project, we were asked to implement the game 2048. The goal of this game is to get, at least, the tile 2048, if not 4096, 8192, etc. The way to get those tiles is to, starting with a random tile of value 2 or 4, in a random cell, to move the tile up, down, to the left or to the right, to get a new tile appear in a random spot. When two adjacent cells have the same tile, they can be merged together (added to form only one tile). By adding more and more tiles we may get the tile 2048. The game was coded in C.

static struct termios old, new;

/* Initialize new terminal i/o settings */
void initTermios(int echo) 
{
  tcgetattr(0, &old); /* grab old terminal i/o settings */
  new = old; /* make new settings same as old settings */
  new.c_lflag &= ~ICANON; /* disable buffered i/o */
  new.c_lflag &= echo ? ECHO : ~ECHO; /* set echo mode */
  tcsetattr(0, TCSANOW, &new); /* use these new terminal i/o settings now */
}

/* Restore old terminal i/o settings */
void resetTermios(void) 
{
  tcsetattr(0, TCSANOW, &old);
}

/* Read 1 character - echo defines echo mode */
char getch_(int echo) 
{
  char ch;
  initTermios(echo);
  ch = getchar();
  resetTermios();
  return ch;
}

/* Read 1 character without echo */
char getch(void) 
{
  return getch_(0);
}

/* Read 1 character with echo */
char getche(void) 
{
  return getch_(1);
}

game * make_game(int rows, int cols)
/*! Create an instance of a game structure with the given number of rows
    and columns, initializing elements to -1 and return a pointer
    to it. (See game.h for the specification for the game data structure) 
    The needed memory should be dynamically allocated with the malloc family
    of functions.
*/
{
    //Dynamically allocate memory for game and cells (DO NOT modify this)
    	game * mygame = malloc(sizeof(game));
    	mygame->cells = malloc(rows*cols*sizeof(cell));

    	//Initializing variables in game struct
	mygame->rows = rows;
	mygame->cols = cols;
	mygame->score = 0;

	//Setting all cells value to -1
	int tot_cells = rows * cols;
	for(int i = 0; i < tot_cells; i++)
	{
		mygame->cells[i] = -1;
	}//end for

    return mygame;
}//end make_game

void remake_game(game ** _cur_game_ptr,int new_rows,int new_cols)
/*! Given a game structure that is passed by reference, change the
	game structure to have the given number of rows and columns. Initialize
	the score and all elements in the cells to -1. Make sure that any 
	memory previously allocated is not lost in this function.	
*/
{
	//Making double pointer a pointer of type game
	game* my_game_ptr = *_cur_game_ptr;	

	/*Frees dynamically allocated memory used by cells in previous game,
	 then dynamically allocates memory for cells in new game.  DO NOT MODIFY.*/
	free((*_cur_game_ptr)->cells);
	(*_cur_game_ptr)->cells = malloc(new_rows*new_cols*sizeof(cell));

	//Re-initializing variables in game struct
	my_game_ptr->rows = new_rows;
	my_game_ptr->cols = new_cols;
	my_game_ptr->score = 0;

	//Re-assigning cells value of -1
	int new_tot_cells = new_rows * new_cols;
	for(int j = 0; j < new_tot_cells; j++)
	{
		my_game_ptr->cells[j] = -1;
	}//end for 

	return;	
}//end remake_game

void destroy_game(game * cur_game)
/*! Deallocate any memory acquired with malloc associated with the given game instance.
    This includes any substructures the game data structure contains. Do not modify this function.*/
{
    free(cur_game->cells);
    free(cur_game);
    cur_game = NULL;
    return;
}//end detroy_game

cell * get_cell(game * cur_game, int row, int col)
/*! Given a game, a row, and a column, return a pointer to the corresponding
    cell on the game. (See game.h for game data structure specification)
    This function should be handy for accessing game cells. Return NULL
	if the row and col coordinates do not exist.
*/
{
    	//Creating new variables
    	int cell_num;
	
	//Checking if row and cell coordinates exist, returning NULL if they do not
	//If coordinates exist sets cell_num value to index value, sets cell_ptr value
	//to mem location value and returns it
	if(row < 0 || col < 0 || row >= cur_game->rows || col >= cur_game->cols)
    		return NULL;
	
	else
		cell_num = col + (row * cur_game->cols);
		cell * cell_ptr = cell_num + cur_game->cells;
		return cell_ptr;
}//end get_cell

int move_w(game * cur_game)
/*!Slides all of the tiles in cur_game upwards. If a tile matches with the 
   one above it, the tiles are merged by adding their values together. When
   tiles merge, increase the score by the value of the new tile. A tile can 
   not merge twice in one turn. If sliding the tiles up does not cause any 
   cell to change value, w is an invalid move and return 0. Otherwise, return 1. 
*/
{
    	//Declaring and initializing values
    	valid_move = 0;
	int upper_cell, bottom_cell;
	int x, y;
	int next_cell;

	//x corresponds to the row and j to the column
	//Cells coordinates will be represented this way: (x, y) 
	for (y = 0; y < cur_game->cols; y++)
	{
		for(x = 0; x < cur_game->rows; x++)
		{
			//Getting index of upper_cell using get_cell function
			upper_cell = *get_cell(cur_game, x, y);
			
			//If upper_cell is not empty checks if bottom_cell matches
			if(upper_cell > 0)
			{
				for(next_cell = x + 1; next_cell < cur_game->rows; next_cell++)
				{
					bottom_cell = *get_cell(cur_game, next_cell, y);
					//Continues if bottom_cell is empty and breaks out
					//of loop if bottom_cell and upper_cell are not the same cell.
					//
					//If up and bottom cells match, merge the cells together
					//doubles value of upper_cell and clears bottom_cell. 
					if(bottom_cell < 0)
					{
						continue;
					}//end if

					if(bottom_cell != upper_cell)
					{
						break;
					}//end if

					upper_cell = 2 * upper_cell;
					bottom_cell = -1;

					*get_cell(cur_game, x, y) = upper_cell;
					*get_cell(cur_game, next_cell, y) = bottom_cell;

					x = next_cell - 1;
		
					//Updating score
					cur_game->score = cur_game->score + upper_cell;
					//Setting valid_move to true
					valid_move = 1;
					//breaks out of next_cell for loop to check for next row
					break;
				}//end for next_cell
			}//end if
		}//end for x
	}//end for y

	//Checking if values can be shifted to upper cells
	for(y = 0; y < cur_game->cols; y++)
	{
		for(x = 0; x < cur_game->rows; x++)
		{
			//Getting upper_cell index using get_cell function
			upper_cell = *get_cell(cur_game, x, y);
	
			//Checking if upper_cell is empty, if it is, going to for loop	
			if(upper_cell < 0)
			{
				for(next_cell = x + 1; next_cell < cur_game->rows; next_cell++)
				{
					bottom_cell = *get_cell(cur_game, next_cell, y);

					if(bottom_cell < 0)
					{
						continue;
					}//end if

					//Shifting value of bottom_cell to upper_cell and clearing bottom_cell	
					upper_cell = bottom_cell;
					bottom_cell = -1;

					//Re-positioning upper and bottom cells 
					*get_cell(cur_game, x, y) = upper_cell;
					*get_cell(cur_game, next_cell, y) = bottom_cell;

					//Setting valid_move to true
					valid_move = 1;
					//breaks out of loop
					break;
				}//end for next_cell
			}//end if
		}//end for x
	}//end for y
	
	//Returning 1 if it is a valid move and a 0 if not
	if(valid_move == 1)	
    		return 1;

	return 0;
};//end move_w

int move_s(game * cur_game) //slide down
{
    	//Declaring and initializing variables
        valid_move = 0;
        int upper_cell, bottom_cell;
        int x, y;
        int next_cell;

	//x corresponds to the row number, and y to the column number
	for(y = cur_game->cols - 1; y >= 0; y--)
	{
		for(x = cur_game->rows - 1; x >= 0; x--)
		{
			//Getting index of bottom_cell using get_cell function
			bottom_cell = *get_cell(cur_game, x, y);

			//Checking if bottom_cell is empty. If not, checking if
			//bottom and upper cells match.
			if(bottom_cell > 0)
			{
				for(next_cell = x - 1; next_cell >= 0; next_cell--)
				{
					upper_cell = *get_cell(cur_game, next_cell, y);

					//Program continues if upper_cell is empty and breaks out
					//out of the loop if bottom and upper cells are different.
					//
					//If up and bottom have same value, merge the cells together
					//towards the bottom, doubles bottom_cell and clears upper_cell.
					if(upper_cell < 0)
					{
						continue;
					}//end if
					if(upper_cell != bottom_cell)
					{
						break;
					}//end if

					bottom_cell = 2 * bottom_cell;
					upper_cell = -1;

					//re_assigning value to main cell and next cell
					*get_cell(cur_game, x, y) = bottom_cell;
					*get_cell(cur_game, next_cell, y) = upper_cell;

					x = next_cell + 1;

					//Updating score
					cur_game->score = cur_game->score + bottom_cell;
					//Setting valid_move to true
					valid_move = 1;
					//Breaking out of the loop
					break;
				}//end for next_cell
			}//end if
		}//end for x
	}//end for y

	//Checking if cells can be shifted down
	for(y = cur_game->cols - 1; y >= 0; y--)
	{
		for(x = cur_game->rows - 1; x >= 0; x--)
		{
			//Getting bottom_cell index using get_cell function
			bottom_cell = *get_cell(cur_game, x, y);

			if(bottom_cell < 0)
			{
				for(next_cell = x - 1; next_cell >= 0; next_cell--)
				{
					upper_cell = *get_cell(cur_game, next_cell, y);

					//Continues even if upper_cell is empty
					if(upper_cell < 0)
					{
						continue;
					}//end if

					//Shifting value of upper_cell to bottom_cell and clears upper_cell
					bottom_cell = upper_cell;
					upper_cell = -1;

					//Re-positioning upper and botton cells
					*get_cell(cur_game, x, y) = bottom_cell;
					*get_cell(cur_game, next_cell, y) = upper_cell;

					//Setting valid_move to true
					valid_move = 1;
					//Breaking out of loop
					break;
				}//end for next_cell
			}//end if	
		}//end for x
	}//end for y
	
	//Returning 1 if valid move
	if(valid_move == 1)
    		return 1;
	else
		return 0;
}; //end move_s

int legal_move_check(game * cur_game)
/*! Given the current game check if there are any legal moves on the board. There are
    no legal moves if sliding in any direction will not cause the game to change.
	Return 1 if there are possible legal moves, 0 if there are none.
 */
{
    	//Declaring and initializing variables
    	int x, y;
	int main_cell, left_cell, right_cell, above_cell, under_cell;	

	for(x = 0; x < cur_game->rows; x++)
	{
		for(y = 0; y < cur_game->cols; y++)
		{
			//Getting value of main_cell and of the 4 cells around it
			main_cell = *get_cell(cur_game, x, y);
			left_cell = *get_cell(cur_game, x, y - 1);
			right_cell = *get_cell(cur_game, x, y + 1);
			above_cell = *get_cell(cur_game, x - 1, y);
			under_cell = *get_cell(cur_game, x + 1, y);

			//If main_cell is empty, there is a possible move so returns 1
			if(main_cell < 0)
				return 1;

			//If main_cell and a cell around it have the same tile, there is
			//a possible move so returns 1
			else if((main_cell == left_cell) || (main_cell == right_cell) || (main_cell == above_cell) || (main_cell == under_cell))
				return 1;
		}//end for y
	}//end for x

	//Returns 0 if no possible move
    	return 0;
}

void rand_new_tile(game * cur_game)
/*! insert a new tile into a random empty cell. First call rand()%(rows*cols) to get a random value between 0 and (rows*cols)-1.
*/
{
	
	cell * cell_ptr;
    cell_ptr = 	cur_game->cells;
	
    if (cell_ptr == NULL){ 	
        printf("Bad Cell Pointer.\n");
        exit(0);
    }
	
	
	//check for an empty cell
	int emptycheck = 0;
	int i;
	
	for(i = 0; i < ((cur_game->rows)*(cur_game->cols)); i++){
		if ((*cell_ptr) == -1){
				emptycheck = 1;
				break;
		}		
        cell_ptr += 1;
	}
	if (emptycheck == 0){
		printf("Error: Trying to insert into no a board with no empty cell. The function rand_new_tile() should only be called after tiles have succesfully moved, meaning there should be at least 1 open spot.\n");
		exit(0);
	}
	
    int ind,row,col;
	int num;
    do{
		ind = rand()%((cur_game->rows)*(cur_game->cols));
		col = ind%(cur_game->cols);
		row = ind/cur_game->cols;
    } while ( *get_cell(cur_game, row, col) != -1);
        //*get_cell(cur_game, row, col) = 2;
	num = rand()%20;
	if(num <= 1){
		*get_cell(cur_game, row, col) = 4; // 1/10th chance
	}
	else{
		*get_cell(cur_game, row, col) = 2;// 9/10th chance
	}
}

int process_turn(const char input_char, game* cur_game) //returns 1 if legal move is possible after input is processed
{ 
	int rows,cols;
	char buf[200];
	char garbage[2];
    int move_success = 0;
	
    switch ( input_char ) {
    case 'w':
        move_success = move_w(cur_game);
        break;
    case 'a':
        move_success = move_a(cur_game);
        break;
    case 's':
        move_success = move_s(cur_game);
        break;
    case 'd':
        move_success = move_d(cur_game);
        break;
    case 'q':
        destroy_game(cur_game);
        printf("\nQuitting..\n");
        return 0;
        break;
	case 'n':
		//get row and col input for new game
		dim_prompt: printf("NEW GAME: Enter dimensions (rows columns):");
		while (NULL == fgets(buf,200,stdin)) {
			printf("\nProgram Terminated.\n");
			return 0;
		}
		
		if (2 != sscanf(buf,"%d%d%1s",&rows,&cols,garbage) ||
		rows < 0 || cols < 0){
			printf("Invalid dimensions.\n");
			goto dim_prompt;
		} 
		
		remake_game(&cur_game,rows,cols);
		
		move_success = 1;
		
    default: //any other input
        printf("Invalid Input. Valid inputs are: w, a, s, d, q, n.\n");
    }

	
	
	
    if(move_success == 1){ //if movement happened, insert new tile and print the game.
         rand_new_tile(cur_game); 
		 print_game(cur_game);
    } 

    if( legal_move_check(cur_game) == 0){  //check if the newly spawned tile results in game over.
        printf("Game Over!\n");
        return 0;
    }
    return 1;
}
