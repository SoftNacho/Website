 * MP/LAB:       MP7: For the machine problem this week, we implemented a sudoku solver.
 *			The way it works very straight forward and uses recursion. There 
 *			are a total of 5 functions that we had to implement. The first and 
 * 			second functions, respectively, check if the value entered is already
 *			in the row or the column of the cell we are looking at. The third 
 *			function checks if that same value has already been used in the 3x3 
 *			zone the cell is in. The fourth function, is_val_valid calls the previous
 *			three functions for a specific row i, column j and value val. If the 
 * 			value passes all three checks, it assigned to the cell. Finally, 
 *			the solve_sudoku function, first checks if all the cells are filled, 
 *			when a non-filled cell is detected it will call is_val_valid which calls
 *			the other functions, and if all the cells are filled, the program stops and
 *			prompts the input sudoku and its solution.
 * TERM:         Spring 2016

// Procedure: print_sudoku
void print_sudoku(int sudoku[9][9])
{
  int i, j;
  for(i=0; i<9; i++) {
    for(j=0; j<9; j++) {
      printf("%2d", sudoku[i][j]);
    }
    printf("\n");
  }
}

// Procedure: parse_sudoku
void parse_sudoku(const char fpath[], int sudoku[9][9]) {
  FILE *reader = fopen(fpath, "r");
  assert(reader != NULL);
  int i, j;
  for(i=0; i<9; i++) {
    for(j=0; j<9; j++) {
      fscanf(reader, "%d", &sudoku[i][j]);
    }
  }
  fclose(reader);
}

// Function: is_val_in_row
// Return true if "val" already existed in ith row of array sudoku.
int is_val_in_row(const int val, const int i, const int sudoku[9][9])
{

  	assert(i>=0 && i<9);
	
	for(int j = 0; j < 9; j++)
	{
		if(sudoku[i][j] == val)
			return 1;
	}//end for
  
  
  	return 0;

}//end is_val_in_row

// Function: is_val_in_col
// Return true if "val" already existed in jth column of array sudoku.
int is_val_in_col(const int val, const int j, const int sudoku[9][9]) 
{

  	assert(j>=0 && j<9);

	for(int i = 0; i < 9; i++)
	{
		if(sudoku[i][j] == val)
			return 1;
	}//end for

  	return 0;

}//end is_val_in_col

/*******************************************************************************************/

// Function: is_val_in_3x3_zone
// Return true if val already existed in the 3x3 zone corresponding to (i, j)
int is_val_in_3x3_zone(const int val, const int i, const int j, const int sudoku[9][9]) 
{
   
  	assert(i>=0 && i<9);

	int ii, jj;
	//ii corresponds to the center row index of the 3x3 zone
	//jj corresponds to the center col index of the 3x3 zone
	//assigning ii and jj values depending on i and j values
	if((i >= 0) && (i <= 2))
		ii = 1;

        if((i >= 3) && (i <= 5))
                ii = 4;

        if((i >= 6) && (i <= 8))
                ii = 7;

        if((j >= 0) && (j <= 2))
           	jj = 1;
                
        if((j >= 3) && (j <= 5))
             	jj = 4;
                
        if((j >= 6) && (j <= 8))
             	jj = 7;

	//checking the 3x3 zone in which (i,j) is included
      	for(int x = ii-1; x <= ii+1; x++)
        {
         	for(int y = jj-1; y <= jj+1; y++)
              	{
                   	if(sudoku[x][y] == val)
                     	       return 1;
       
            	}//end for y
    	}//end for x

  	return 0;

}//end is_val_in_3x3_zone

/*******************************************************************************************/

// Function: is_val_valid
// Return true if the val is can be filled in the given entry.
int is_val_valid(const int val, const int i, const int j, const int sudoku[9][9]) 
{

	int row_check = is_val_in_row(val, i, sudoku);
	int col_check = is_val_in_col(val, j, sudoku);
	int zone_check = is_val_in_3x3_zone(val, i, j, sudoku); 

	//returns 1 if the input value is valid
  	if((row_check == 0) && (col_check == 0) && (zone_check == 0))
		return 1; 
  
	else
		return 0;

}//end is_val_valid

// Procedure: solve_sudoku
// Solve the given sudoku instance.
int solve_sudoku(int sudoku[9][9]) 
{
	//declaring new variables
	int i, j, p, q;
	int non_filled_found;
	
	// check if there is any non-filled cells. If there is, ut breaks out of
	// of both for loops (i and j), find a value that could fill the cell,
	// assign this value to the cell, calls the solve_sudoku function (recursion)	
	for(i = 0; i < 9; i++)
	{
		for(j = 0; j < 9; j++)
		{
			if(sudoku[i][j] == 0)
			{
				non_filled_found = 1;
				break;
			}//end if

		}//end for j
		
		if (non_filled_found == 1)
			break;
	}//end for i 
	
	if(non_filled_found == 0)
		return 1;

	for(int num = 1; num <= 9; num++)
	{
		if(is_val_valid(num, i, j, sudoku) == 1)
		{
			sudoku[i][j] = num;

			if(solve_sudoku(sudoku))
			{
				return 1;
			}//end if
			
			sudoku[i][j] = 0;
		}//end if
	}//end for num

  return 0;
}//end solve_sudoku
